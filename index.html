Sample: Simple Python revenue projection


# revenue_projection.py
def project_revenue(base_rev, growth_rate, years=3):
    projections = []
    rev = base_rev
    for y in range(1, years+1):
        rev = rev * (1 + growth_rate)
        projections.append((y, round(rev, 2)))
    return projections

print(project_revenue(1_000_000, 0.08, years=4))
